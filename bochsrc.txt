#========================================================================
# BOCHSRC - Konfigurasi Emulator Bochs untuk OS buatan sendiri
#========================================================================

# -----------------------------------------------------------------------
# ROM BIOS dan VGA BIOS (gunakan path absolut)
# -----------------------------------------------------------------------
romimage: file="/usr/share/bochs/BIOS-bochs-legacy"
vgaromimage: file="/usr/share/bochs/VGABIOS-lgpl-latest"

# -----------------------------------------------------------------------
# CPU dan clock rate
# -----------------------------------------------------------------------
cpu: count=1, ips=10000000, reset_on_triple_fault=1

# -----------------------------------------------------------------------
# RAM 32 MB
# -----------------------------------------------------------------------
megs: 32

# -----------------------------------------------------------------------
# Tipe VGA dan frekuensi refresh
# -----------------------------------------------------------------------
vga: extension=vbe, update_freq=10

# -----------------------------------------------------------------------
# Floppy boot image (gunakan floppy.img buatan sendiri)
# -----------------------------------------------------------------------
floppya: 1_44="bin/floppy.img", status=inserted

# -----------------------------------------------------------------------
# Urutan boot: floppy dulu
# -----------------------------------------------------------------------
boot: floppy

# -----------------------------------------------------------------------
# Cek signature floppy boot (AA55h)
# -----------------------------------------------------------------------
floppy_bootsig_check: disabled=0

# -----------------------------------------------------------------------
# Log output (ke terminal)
# -----------------------------------------------------------------------
log: -
debugger_log: -

# -----------------------------------------------------------------------
# Level logging
# -----------------------------------------------------------------------
panic: action=ask
error: action=report
info: action=report
debug: action=ignore

# -----------------------------------------------------------------------
# Mouse nonaktif (tidak diperlukan)
# -----------------------------------------------------------------------
mouse: enabled=0

# -----------------------------------------------------------------------
# PCI aktif (jika ingin gunakan device tambahan di masa depan)
# -----------------------------------------------------------------------
pci: enabled=1, chipset=i440fx
